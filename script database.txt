Criar a tabela contatos

create table contatos(
	id serial primary key,
	nome varchar(40),
	idade int,
	dataCadastro date
)

create table telefones(
	id serial primary key,
	telefone int,
	contato_id int,
	FOREIGN KEY (contato_id) REFERENCES contatos (id)
)
